    """Get a patient using the API"""
    url = URL + patient_code + "/"+ str(id)
    try:
        response = requests.get(url)
        json_data = response.json()

        print("Patient retrieve")
        return patient
    except:

        return None

    """This function is used to add patient to the database"""

        return False

    """
